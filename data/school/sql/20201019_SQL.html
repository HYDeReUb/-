使用store_information資料庫

--在store_information中選出所有Sales高於$1000或是Sales在$500及$275之間的資料

SELECT Store_Name FROM `store_information` WHERE Sales > 1000 OR (Sales < 500 AND Sales > 275)

--秀出指定名稱的資訊

SELECT * FROM `store_information` WHERE Store_Name IN ('Los Angeles', 'San Diego')

--秀出有A或a的名稱

SELECT * FROM store_information WHERE Store_Name LIKE '%A%'

--秀出有A或a的名稱筆數

SELECT COUNT(*) AS 筆數 FROM store_information WHERE Store_Name LIKE '%A%'

--尋找2個值中間的資料

SELECT * FROM `store_information` WHERE Txn_Date BETWEEN '2019-10-06' AND '2019-10-08'

--以Sales為準由高往低排列

SELECT Store_Name, Sales, Txn_Date FROM Store_Information ORDER BY Sales DESC

使用student_detail資料庫

--以std_grade為準並不小於70分，由高往低排列

SELECT std_name, std_grade FROM student_detail WHERE std_grade >= 70 ORDER BY std_grade DESC
